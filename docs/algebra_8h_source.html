434 inline static void lcg_dot(lcg_complex &ret, const lcg_complex *a,
435  const lcg_complex *b, int size)
436 {
437  ret.set(0.0, 0.0);
438  // <a,b> = \sum{a_i \cdot b_i}
439  for (int i = 0; i < size; i++)
440  {
441  ret.rel += (a[i].rel*b[i].rel - a[i].img*b[i].img);
442  ret.img += (a[i].rel*b[i].img + a[i].img*b[i].rel);
443  }
444  return;
445 }
